class TemperatureTooHighError(Exception):
    """
    Raised when we attempt to consume the moment before it is ready.

    """

class Cup:
    """

    """
    def __init__(self, coffee_type="Dark Roast of Clarity"):
        # We manifest the object. It begins full, hot, and full of promise.

        self.fill_level = 100  # Percentage
        self.temperature = 95  # Celsius. A dangerous heat.
        print(f"--> You have encompassed a warm mug of {self.contents}.")

    def _cool_down(self):
        # Entropy is natural. We let the universe take the heat.
        self.temperature -= 15

    def sip(self):
        # An attempt to intake energy.
        print("\n... Lifting the cup ...")
        
        # Check the conditions. Is the world too intense right now?
        if self.temperature > 85:
            # If we rush, we suffer. Raise the alarm.
            raise TemperatureTooHighError()

    def refill(self):
        # The eternal cycle of renewal.
